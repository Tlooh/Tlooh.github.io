import os
from openai import OpenAI

os.environ["OPENAI_API_KEY"] = "sk---xxxxxx---xxxxxx" # 添加环境变量
# api_key = "sk---xxxxxx---xxxxxx" # 或者直接传入
# api_base_url = "https://xxx.xxx.com/" # 如果是非官方渠道的 api_key 才需要提供

client = OpenAI(
    api_key=os.environ.get("OPENAI_API_KEY"),
    # base_url = base_api_url # 如果是非官方渠道的 api_key 才需要提供
)


chat_completion = client.chat.completions.create(
    messages=[
        {
            "role": "user",
            "content": "Say this is a test",
        }
    ],
    model="gpt-3.5-turbo",
)

return chat_completion.choices[0].message.content

3. 参数解释

（1）“message"消息列表

message 是一个列表，每个元素都是一个字典，包含"role"和"content"两个键。

  • “role"用于指定发言的角色，
  • “content"则包含该角色的具体发言内容。

通过在"message"参数中交替指定不同角色的发言，可以模拟对话或交互式场景。模型将根据之前的对话历史和当前的角色发言生成回应。 在OpenAI API中，“role"字段可以根据对话场景的需求指定不同的角色。具体角色的选择可以根据对话的上下文和应用场景来决定。以下是一些常见的角色及其示例：

用户（User）：代表与系统进行对话的用户。用户角色通常提出问题、提供指令或表达需求。

example：{“role”: “user”, “content”: “请问明天的天气如何？”}

助手（Assistant）：代表系统的智能助手或回答提供者。助手角色负责根据用户的需求提供回答、建议或解决方案。

example：{“role”: “assistant”, “content”: “明天的天气预报显示晴朗，最高温度为25摄氏度。”}

系统（System）：代表整个对话系统或指导者角色。系统角色可以提供对话的初始设置、说明或引导用户与助手的交互流程。

示例：{“role”: “system”, “content”: “欢迎使用天气查询系统，请告诉我您想要查询哪个城市的天气。”}

（2）“temperature”

它是一个介于0和1之间的浮点数，其中较小的值会使生成的文本更加保守和确定性，而较大的值则会使文本更加多样化和随机。

  • 当使用较小的temperature值（例如0.2或0.5）时，模型更倾向于选择那些高概率的词汇和短语，生成的文本更加保守和可预测。这通常会导致生成的文本更加流畅，但可能会缺乏创造性。
  • 当使用较大的temperature值（例如1.0或1.2）时，模型更容易选择低概率的词汇和短语，生成的文本更加多样化和随机。这可能会导致一些不合理或不通顺的文本，但也可能带来更具创意和想象力的结果。

（3）“top_p”

“Topp"参数是一个介于0和1之间的浮点数(不能等于0)，表示累积概率阈值。在生成文本时，模型会根据每个词的概率进行排序,并选择累积概率达到或超过"Topp"阈值的最小词汇集。换句话说，它限制了模型从高概率词汇中进行选择的范围。

（4）“max_tokens”

参数用于限制生成文本的最大长度，以控制生成结果的长度。

4. Python + openai 实现多轮对话

里面传入的 message 其实是一个列表，列表里每一项其实是一个字典，包含了角色和内容，我们只需要在提问的时候，把回答也加入进来更新 messages，也就是每次提问都带上之前的问题和回答；

import os
from openai import OpenAI

def get_response(messages, model="gpt-3.5-turbo", temperature=1, top_p=1, max_tokens=1000):

    chat_completion = client.chat.completions.create(
        messages=messages,
        temperature=temperature,
        top_p=top_p,
        max_tokens=max_tokens,
        model=model,
    )
    
    return chat_completion.choices[0].message.content

if __name__ == "__main__":
    api_key = "sk---xxxxx---xxxxxx=="
    base_url = "http://xxxx.xxxx.com"

    client = OpenAI(
        # This is the default and can be omitted
        api_key=api_key,
        base_url=base_url
    )
    messages = [
        {"role": "system", "content": "You are a helpful assistant."},
        {"role": "user", "content": "What's the weather like today?"},
    ]
    anser = get_response(messages)
    print(f"[ChatGPT]" + anser)
    gpt_anser = {"role": "assistant", "content": anser}
    messages.append(gpt_anser)
    query =  {"role": "user", "content": "That sounds great! What do you recommend I should eat for breakfast?"}
    messages.append(query)
    anser = get_response(messages)
    print(f"[ChatGPT]" + anser)
